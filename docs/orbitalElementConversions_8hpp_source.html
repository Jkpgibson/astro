67 template< typename Real, typename Vector6 >
68 Vector6 convertCartesianToKeplerianElements(
69  const Vector6& cartesianElements,
70  const Real gravitationalParameter,
71  const Real tolerance = 10.0 * std::numeric_limits< Real >::epsilon( ) )
72 {
73  // Check that the Cartesian elements vector contains exactly 6 elemenets and otherwise throw
74  // and error.
75  if ( cartesianElements.size( ) != 6 )
76  {
77  throw std::runtime_error(
78  "ERROR: Cartesian elements vector has more or less than 6 elements!" );
79  }
80 
81  Vector6 keplerianElements = cartesianElements;
82 
83  typedef std::vector< Real > Vector;
84 
85  // Set position and velocity vectors.
86  Vector position( 3 );
87  position[ xPositionIndex ] = cartesianElements[ xPositionIndex ];
88  position[ yPositionIndex ] = cartesianElements[ yPositionIndex ];
89  position[ zPositionIndex ] = cartesianElements[ zPositionIndex ];
90 
91  Vector velocity( 3 );
92  velocity[ 0 ] = cartesianElements[ xVelocityIndex ];
93  velocity[ 1 ] = cartesianElements[ yVelocityIndex ];
94  velocity[ 2 ] = cartesianElements[ zVelocityIndex ];
95 
96  // Compute orbital angular momentum vector.
97  Vector angularMomentum( 3 );
98  angularMomentum[ 0 ] = position[ 1 ] * velocity[ 2 ] - position[ 2 ] * velocity[ 1 ];
99  angularMomentum[ 1 ] = position[ 2 ] * velocity[ 0 ] - position[ 0 ] * velocity[ 2 ];
100  angularMomentum[ 2 ] = position[ 0 ] * velocity[ 1 ] - position[ 1 ] * velocity[ 0 ];
101 
102  // Compute semi-latus rectum.
103  const Real angularMomentumSquaredNorm = angularMomentum[ 0 ] * angularMomentum[ 0 ]
104  + angularMomentum[ 1 ] * angularMomentum[ 1 ]
105  + angularMomentum[ 2 ] * angularMomentum[ 2 ];
106  const Real semiLatusRectum = angularMomentumSquaredNorm / gravitationalParameter;
107 
108  // Compute unit vector to ascending node.
109  const Real angularMomentumNorm = std::sqrt( angularMomentumSquaredNorm );
110  Vector angularMomentumNormalized( 3 );
111  angularMomentumNormalized[ 0 ] = angularMomentum[ 0 ] / angularMomentumNorm;
112  angularMomentumNormalized[ 1 ] = angularMomentum[ 1 ] / angularMomentumNorm;
113  angularMomentumNormalized[ 2 ] = angularMomentum[ 2 ] / angularMomentumNorm;
114 
115  const Vector zUnitVector( { 0.0, 0.0, 1.0 } );
116 
117  Vector ascendingNodeVector( 3 );
118  ascendingNodeVector[ 0 ] = zUnitVector[ 1 ] * angularMomentumNormalized[ 2 ]
119  - zUnitVector[ 2 ] * angularMomentumNormalized[ 1 ];
120  ascendingNodeVector[ 1 ] = zUnitVector[ 2 ] * angularMomentumNormalized[ 0 ]
121  - zUnitVector[ 0 ] * angularMomentumNormalized[ 2 ];
122  ascendingNodeVector[ 2 ] = zUnitVector[ 0 ] * angularMomentumNormalized[ 1 ]
123  - zUnitVector[ 1 ] * angularMomentumNormalized[ 0 ];
124 
125  const Real ascendingNodeVectorNorm
126  = std::sqrt( ascendingNodeVector[ 0 ] * ascendingNodeVector[ 0 ]
127  + ascendingNodeVector[ 1 ] * ascendingNodeVector[ 1 ]
128  + ascendingNodeVector[ 2 ] * ascendingNodeVector[ 2 ] );
129 
130  Vector ascendingNodeUnitVector( 3 );
131  ascendingNodeUnitVector[ 0 ] = ascendingNodeVector[ 0 ] / ascendingNodeVectorNorm;
132  ascendingNodeUnitVector[ 1 ] = ascendingNodeVector[ 1 ] / ascendingNodeVectorNorm;
133  ascendingNodeUnitVector[ 2 ] = ascendingNodeVector[ 2 ] / ascendingNodeVectorNorm;
134 
135  // Compute eccentricity vector.
136  const Real positionNorm = std::sqrt( position[ 0 ] * position[ 0 ]
137  + position[ 1 ] * position[ 1 ]
138  + position[ 2 ] * position[ 2 ] );
139  Vector positionNormalized( 3 );
140  positionNormalized[ 0 ] = position[ 0 ] / positionNorm;
141  positionNormalized[ 1 ] = position[ 1 ] / positionNorm;
142  positionNormalized[ 2 ] = position[ 2 ] / positionNorm;
143 
144  Vector eccentricityVector( 3 );
145  eccentricityVector[ 0 ] = ( velocity[ 1 ] * angularMomentum[ 2 ]
146  - velocity[ 2 ] * angularMomentum[ 1 ] ) / gravitationalParameter
147  - positionNormalized[ 0 ];
148  eccentricityVector[ 1 ] = ( velocity[ 2 ] * angularMomentum[ 0 ]
149  - velocity[ 0 ] * angularMomentum[ 2 ] ) / gravitationalParameter
150  - positionNormalized[ 1 ];
151  eccentricityVector[ 2 ] = ( velocity[ 0 ] * angularMomentum[ 1 ]
152  - velocity[ 1 ] * angularMomentum[ 0 ] ) / gravitationalParameter
153  - positionNormalized[ 2 ];
154 
155  // Store eccentricity.
156  const Real eccentricityVectorNorm
157  = std::sqrt( eccentricityVector[ 0 ] * eccentricityVector[ 0 ]
158  + eccentricityVector[ 1 ] * eccentricityVector[ 1 ]
159  + eccentricityVector[ 2 ] * eccentricityVector[ 2 ] );
160 
161  Vector eccentricityVectorNormalized( 3 );
162  eccentricityVectorNormalized[ 0 ] = eccentricityVector[ 0 ] / eccentricityVectorNorm;
163  eccentricityVectorNormalized[ 1 ] = eccentricityVector[ 1 ] / eccentricityVectorNorm;
164  eccentricityVectorNormalized[ 2 ] = eccentricityVector[ 2 ] / eccentricityVectorNorm;
165 
166  keplerianElements[ eccentricityIndex ] = eccentricityVectorNorm;
167 
168  // Compute and store semi-major axis.
169  // Check if orbit is parabolic. If it is, store the semi-latus rectum instead of the
170  // semi-major axis.
171  if ( std::fabs( keplerianElements[ eccentricityIndex ] - 1.0 ) < tolerance )
172  {
173  keplerianElements[ semiLatusRectumIndex ] = semiLatusRectum;
174  }
175 
176  // Else the orbit is either elliptical or hyperbolic, so store the semi-major axis.
177  else
178  {
179  keplerianElements[ semiMajorAxisIndex ]
180  = semiLatusRectum / ( 1.0 - keplerianElements[ eccentricityIndex ]
181  * keplerianElements[ eccentricityIndex ] );
182  }
183 
184  // Compute and store inclination.
185  keplerianElements[ inclinationIndex ]
186  = std::acos( angularMomentum[ zPositionIndex ] / angularMomentumNorm );
187 
188  // Compute and store longitude of ascending node.
189  // Define the quadrant condition for the argument of perigee.
190  Real argumentOfPeriapsisQuandrantCondition = eccentricityVector[ zPositionIndex ];
191 
192  // Check if the orbit is equatorial. If it is, set the vector to the line of nodes to the
193  // x-axis.
194  if ( std::fabs( keplerianElements[ inclinationIndex ] ) < tolerance )
195  {
196  const Vector ascendingNodeUnitVector( { 1.0, 0.0, 0.0 } );
197 
198  // If the orbit is equatorial, eccentricityVector_z is zero, therefore the quadrant
199  // condition is taken to be the y-component, eccentricityVector_y.
200  argumentOfPeriapsisQuandrantCondition = eccentricityVector[ yPositionIndex ];
201  }
202 
203  // Compute and store the resulting longitude of ascending node.
204  keplerianElements[ longitudeOfAscendingNodeIndex ]
205  = std::acos( ascendingNodeUnitVector[ xPositionIndex ] );
206 
207  // Check if the quandrant is correct.
208  if ( ascendingNodeUnitVector[ yPositionIndex ] < 0.0 )
209  {
210  keplerianElements[ longitudeOfAscendingNodeIndex ]
211  = 2.0 * 3.14159265358979323846 - keplerianElements[ longitudeOfAscendingNodeIndex ];
212  }
213 
214  // Compute and store argument of periapsis.
215  // Define the quadrant condition for the true anomaly.
216  Real trueAnomalyQuandrantCondition = position[ 0 ] * velocity[ 0 ]
217  + position[ 1 ] * velocity[ 1 ]
218  + position[ 2 ] * velocity[ 2 ];
219 
220  // Check if the orbit is circular. If it is, set the eccentricity vector to unit vector
221  // pointing to the ascending node, i.e. set the argument of periapsis to zero.
222  if ( std::fabs( keplerianElements[ eccentricityIndex ] ) < tolerance )
223  {
224  eccentricityVector = ascendingNodeUnitVector;
225 
226  keplerianElements[ argumentOfPeriapsisIndex ] = 0.0;
227 
228  // Check if orbit is also equatorial and set true anomaly quandrant check condition
229  // accordingly.
230  if ( ascendingNodeUnitVector == Vector( { 1.0, 0.0, 0.0 } ) )
231  {
232  // If the orbit is circular, dot( position, velocity ) = 0, therefore this value
233  // cannot be used as a quadrant condition. Moreover, if the orbit is equatorial,
234  // position_z is also zero and therefore the quadrant condition is taken to be the
235  // y-component, position_y.
236  trueAnomalyQuandrantCondition = position[ yPositionIndex ];
237  }
238 
239  else
240  {
241  // If the orbit is circular, dot( position, velocity ) = 0, therefore the quadrant
242  // condition is taken to be the z-component of the position, position_z.
243  trueAnomalyQuandrantCondition = position[ zPositionIndex ];
244  }
245  }
246 
247  // Else, compute the argument of periapsis as the angle between the eccentricity vector and
248  // the unit vector to the ascending node.
249  else
250  {
251  keplerianElements[ argumentOfPeriapsisIndex ]
252  = std::acos( eccentricityVectorNormalized[ 0 ] * ascendingNodeUnitVector[ 0 ]
253  + eccentricityVectorNormalized[ 1 ] * ascendingNodeUnitVector[ 1 ]
254  + eccentricityVectorNormalized[ 2 ] * ascendingNodeUnitVector[ 2 ] );
255 
256  // Check if the quadrant is correct.
257  if ( argumentOfPeriapsisQuandrantCondition < 0.0 )
258  {
259  keplerianElements[ argumentOfPeriapsisIndex ]
260  = 2.0 * 3.14159265358979323846 - keplerianElements[ argumentOfPeriapsisIndex ];
261  }
262  }
263 
264  // Compute dot-product of position and eccentricity vectors.
265  Real dotProductPositionAndEccentricityVectors
266  = positionNormalized[ 0 ] * eccentricityVectorNormalized[ 0 ]
267  + positionNormalized[ 1 ] * eccentricityVectorNormalized[ 1 ]
268  + positionNormalized[ 2 ] * eccentricityVectorNormalized[ 2 ];
269 
270  // Check if the dot-product is one of the limiting cases: 0.0 or 1.0
271  // (within prescribed tolerance).
272  if ( std::fabs( 1.0 - dotProductPositionAndEccentricityVectors ) < tolerance )
273  {
274  dotProductPositionAndEccentricityVectors = 1.0;
275  }
276 
277  if ( std::fabs( dotProductPositionAndEccentricityVectors ) < tolerance )
278  {
279  dotProductPositionAndEccentricityVectors = 0.0;
280  }
281 
282  // Compute and store true anomaly.
283  keplerianElements[ trueAnomalyIndex ] = std::acos( dotProductPositionAndEccentricityVectors );
284 
285  // Check if the quandrant is correct.
286  if ( trueAnomalyQuandrantCondition < 0.0 )
287  {
288  keplerianElements[ trueAnomalyIndex ]
289  = 2.0 * 3.14159265358979323846 - keplerianElements[ trueAnomalyIndex ];
290  }
291 
292  return keplerianElements;
293 }

325 template< typename Real, typename Vector6 >
326 Vector6 convertKeplerianToCartesianElements(
327  const Vector6& keplerianElements, const Real gravitationalParameter,
328  const Real tolerance = 10.0 * std::numeric_limits< Real >::epsilon( ) )
329 {
330  Vector6 cartesianElements = keplerianElements;
331 
332  const Real semiMajorAxis = keplerianElements[ semiMajorAxisIndex ];
333  const Real eccentricity = keplerianElements[ eccentricityIndex ];
334  const Real inclination = keplerianElements[ inclinationIndex ];
335  const Real argumentOfPeriapsis = keplerianElements[ argumentOfPeriapsisIndex ];
336  const Real longitudeOfAscendingNode = keplerianElements[ longitudeOfAscendingNodeIndex ];
337  const Real trueAnomaly = keplerianElements[ trueAnomalyIndex ];
338 
339  // Pre-compute sines and cosines of angles for efficient computation.
340  const Real cosineOfInclination = std::cos( inclination );
341  const Real sineOfInclination = std::sin( inclination );
342  const Real cosineOfArgumentOfPeriapsis = std::cos( argumentOfPeriapsis );
343  const Real sineOfArgumentOfPeriapsis = std::sin( argumentOfPeriapsis );
344  const Real cosineOfLongitudeOfAscendingNode = std::cos( longitudeOfAscendingNode );
345  const Real sineOfLongitudeOfAscendingNode = std::sin( longitudeOfAscendingNode );
346  const Real cosineOfTrueAnomaly = std::cos( trueAnomaly );
347  const Real sineOfTrueAnomaly = std::sin( trueAnomaly );
348 
349  // Compute semi-latus rectum in the case the orbit is not a parabola.
350  Real semiLatusRectum = 0.0;
351  if ( std::fabs( eccentricity - 1.0 ) > tolerance )
352  {
353  semiLatusRectum = semiMajorAxis * ( 1.0 - eccentricity * eccentricity );
354  }
355 
356  // Else set the semi-latus rectum as the first element in the vector of Keplerian elements.
357  else
358  {
359  semiLatusRectum = keplerianElements[ 0 ];
360  }
361 
362  // Compute the magnitude of the orbital radius, measured from the focal point.
363  const Real radiusMagnitude = semiLatusRectum / ( 1.0 + eccentricity * cosineOfTrueAnomaly );
364 
365  // Define position and velocity in the perifocal coordinate system.
366  const Real xPositionPerifocal = radiusMagnitude * cosineOfTrueAnomaly;
367  const Real yPositionPerifocal = radiusMagnitude * sineOfTrueAnomaly;
368  const Real xVelocityPerifocal
369  = -std::sqrt( gravitationalParameter / semiLatusRectum ) * sineOfTrueAnomaly;
370  const Real yVelocityPerifocal
371  = std::sqrt( gravitationalParameter / semiLatusRectum )
372  * ( eccentricity + cosineOfTrueAnomaly );
373 
374  // Compute scalar components of rotation matrix to rotate from periforcal to Earth-Centered
375  // Inertial (ECI) frame.
376  const Real rotationMatrixComponent11
377  = ( cosineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis
378  - sineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis * cosineOfInclination );
379  const Real rotationMatrixComponent12
380  = ( -cosineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis
381  -sineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis * cosineOfInclination );
382 
383  const Real rotationMatrixComponent21
384  = ( sineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis
385  + cosineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis * cosineOfInclination );
386  const Real rotationMatrixComponent22
387  = ( -sineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis
388  + cosineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis * cosineOfInclination );
389 
390  const Real rotationMatrixComponent31 = ( sineOfArgumentOfPeriapsis * sineOfInclination );
391  const Real rotationMatrixComponent32 = ( cosineOfArgumentOfPeriapsis * sineOfInclination );
392 
393  // Compute Cartesian position and velocities.
394  cartesianElements[ xPositionIndex ] = rotationMatrixComponent11 * xPositionPerifocal
395  + rotationMatrixComponent12 * yPositionPerifocal;
396 
397  cartesianElements[ yPositionIndex ] = rotationMatrixComponent21 * xPositionPerifocal
398  + rotationMatrixComponent22 * yPositionPerifocal;
399 
400  cartesianElements[ zPositionIndex ] = rotationMatrixComponent31 * xPositionPerifocal
401  + rotationMatrixComponent32 * yPositionPerifocal;
402 
403  cartesianElements[ xVelocityIndex ] = rotationMatrixComponent11 * xVelocityPerifocal
404  + rotationMatrixComponent12 * yVelocityPerifocal;
405 
406  cartesianElements[ yVelocityIndex ] = rotationMatrixComponent21 * xVelocityPerifocal
407  + rotationMatrixComponent22 * yVelocityPerifocal;
408 
409  cartesianElements[ zVelocityIndex ] = rotationMatrixComponent31 * xVelocityPerifocal
410  + rotationMatrixComponent32 * yVelocityPerifocal;
411 
412  return cartesianElements;
413 }

429 template< typename Real >
430 Real convertTrueAnomalyToEllipticalEccentricAnomaly( const Real trueAnomaly,
431  const Real eccentricity )
432 {
433  if ( eccentricity >= 1.0 || eccentricity < 0.0 )
434  {
435  throw std::runtime_error( "ERROR: Eccentricity is non-elliptical!" );
436  }
437 
438  // Compute sine and cosine of eccentric anomaly.
439  const Real sineOfEccentricAnomaly
440  = std::sqrt( 1.0 - std::pow( eccentricity, 2.0 ) )
441  * std::sin( trueAnomaly ) / ( 1.0 + eccentricity * std::cos( trueAnomaly ) );
442  const Real cosineOfEccentricAnomaly
443  = ( eccentricity + std::cos( trueAnomaly ) )
444  / ( 1.0 + eccentricity * std::cos( trueAnomaly ) );
445 
446  // Return elliptical eccentric anomaly.
447  return std::atan2( sineOfEccentricAnomaly, cosineOfEccentricAnomaly );
448 }

465 template< typename Real >
466 Real convertTrueAnomalyToHyperbolicEccentricAnomaly( const Real trueAnomaly,
467  const Real eccentricity )
468 {
469  if ( eccentricity <= 1.0 )
470  {
471  throw std::runtime_error( "ERROR: Eccentricity is non-hyperbolic!" );
472  }
473 
474  // Compute hyperbolic sine and hyperbolic cosine of hyperbolic eccentric anomaly.
475  const Real hyperbolicSineOfHyperbolicEccentricAnomaly
476  = std::sqrt( std::pow( eccentricity, 2.0 ) - 1.0 )
477  * std::sin( trueAnomaly ) / ( 1.0 + std::cos( trueAnomaly ) );
478 
479  const Real hyperbolicCosineOfHyperbolicEccentricAnomaly
480  = ( std::cos( trueAnomaly ) + eccentricity ) / ( 1.0 + std::cos( trueAnomaly ) );
481 
482  // Return hyperbolic eccentric anomaly.
483  // The inverse hyperbolic tangent is computed here manually, since the atanh() function is not
484  // available in older C++ compilers.
485  const Real angle
486  = hyperbolicSineOfHyperbolicEccentricAnomaly
487  / hyperbolicCosineOfHyperbolicEccentricAnomaly;
488  return 0.5 * ( std::log( 1.0 + angle ) - std::log( 1.0 - angle ) );
489 }

661 Real convertEllipticalEccentricAnomalyToTrueAnomaly( const Real ellipticEccentricAnomaly,
662  const Real eccentricity )
663 {
664  const Real sineOfTrueAnomaly = std::sqrt( 1.0 - eccentricity * eccentricity )
665  * std::sin( ellipticEccentricAnomaly )
666  / ( 1.0 - eccentricity * std::cos( ellipticEccentricAnomaly ) );
667

669  = ( std::cos( ellipticEccentricAnomaly ) - eccentricity )
