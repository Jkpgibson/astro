68 template< typename Real, typename Vector6 >
69 Vector6 convertCartesianToKeplerianElements(
70  const Vector6& cartesianElements,
71  const Real gravitationalParameter,
72  const Real tolerance = 10.0 * std::numeric_limits< Real >::epsilon( ) )
73 {
74  const Real pi = 3.14159265358979323846;
75 
76  if ( cartesianElements.size( ) != 6 )
77  {
78  std::ostringstream errorMessage;
79  errorMessage << "astro::convertCartesianToKeplerianElements()" << std::endl;
80  errorMessage << " ERROR: Cartesian elements vector has more or less than 6 elements!"
81  << std::endl;
82 
83  throw std::runtime_error( errorMessage.str( ) );
84  }
85 
86  typedef std::vector< Real > Vector;
87  const Vector position = Vector( { cartesianElements[ 0 ],
88  cartesianElements[ 1 ],
89  cartesianElements[ 2 ] } );
90  const Real positionNormSquared = position[ 0 ] * position[ 0 ]
91  + position[ 1 ] * position[ 1 ]
92  + position[ 2 ] * position[ 2 ];
93  const Real positionNorm = std::sqrt( positionNormSquared );
94 
95  const Vector velocity = Vector( { cartesianElements[ 3 ],
96  cartesianElements[ 4 ],
97  cartesianElements[ 5 ] } );
98  const Real velocityNormSquared = velocity[ 0 ] * velocity[ 0 ]
99  + velocity[ 1 ] * velocity[ 1 ]
100  + velocity[ 2 ] * velocity[ 2 ];
101 
102  const Vector angularMomentum
103  = Vector( { position[ 1 ] * velocity[ 2 ] - position[ 2 ] * velocity[ 1 ],
104  position[ 2 ] * velocity[ 0 ] - position[ 0 ] * velocity[ 2 ],
105  position[ 0 ] * velocity[ 1 ] - position[ 1 ] * velocity[ 0 ] } );
106  const Real angularMomentumNormSquared = angularMomentum[ 0 ] * angularMomentum[ 0 ]
107  + angularMomentum[ 1 ] * angularMomentum[ 1 ]
108  + angularMomentum[ 2 ] * angularMomentum[ 2 ];
109  const Real angularMomentumNorm = std::sqrt( angularMomentumNormSquared );
110  const Vector angularMomentumUnitVector
111  = Vector( { angularMomentum[ 0 ] / angularMomentumNorm,
112  angularMomentum[ 1 ] / angularMomentumNorm,
113  angularMomentum[ 2 ] / angularMomentumNorm } );
114 
115  const Real inclination = std::acos( angularMomentumUnitVector[ 2 ] );
116 
117  const Real eccentricityPreMultiplier
118  = 1.0 / gravitationalParameter
119  * ( velocityNormSquared - gravitationalParameter / positionNorm );
120 
121  const Real positionDotVelocity = position[ 0 ] * velocity[ 0 ]
122  + position[ 1 ] * velocity[ 1 ]
123  + position[ 2 ] * velocity[ 2 ];
124 
125  const Vector eccentricityVector
126  = Vector( { eccentricityPreMultiplier * position[ 0 ]
127  - positionDotVelocity * velocity[ 0 ] / gravitationalParameter,
128  eccentricityPreMultiplier * position[ 1 ]
129  - positionDotVelocity * velocity[ 1 ] / gravitationalParameter,
130  eccentricityPreMultiplier * position[ 2 ]
131  - positionDotVelocity * velocity[ 2 ] / gravitationalParameter } );
132 
133  const Real eccentricityNormSquared = eccentricityVector[ 0 ] * eccentricityVector[ 0 ]
134  + eccentricityVector[ 1 ] * eccentricityVector[ 1 ]
135  + eccentricityVector[ 2 ] * eccentricityVector[ 2 ];
136 
137  const Real eccentricity = std::sqrt( eccentricityNormSquared );
138 
139  const Real semiMajorAxis
140  = ( std::fabs( eccentricity - 1.0 ) < tolerance )
141  ? angularMomentumNormSquared / gravitationalParameter
142  : 1.0 / ( 2.0 / positionNorm - velocityNormSquared / gravitationalParameter );
143 
144  const Vector ascendingNodeVector
145  = Vector( { -angularMomentumUnitVector[ 1 ], angularMomentumUnitVector[ 0 ], 0.0 } );
146 
147  const Real ascendingNodeVectorNormSquared
148  = ascendingNodeVector[ 0 ] * ascendingNodeVector[ 0 ]
149  + ascendingNodeVector[ 1 ] * ascendingNodeVector[ 1 ]
150  + ascendingNodeVector[ 2 ] * ascendingNodeVector[ 2 ];
151 
152  const Real ascendingNodeVectorNorm = std::sqrt( ascendingNodeVectorNormSquared );
153 
154  const Vector ascendingNodeUnitVector
155  = Vector( { ascendingNodeVector[ 0 ] / ascendingNodeVectorNorm,
156  ascendingNodeVector[ 1 ] / ascendingNodeVectorNorm,
157  ascendingNodeVector[ 2 ] / ascendingNodeVectorNorm } );
158 
159  Real longitudeOfAscendingNode = 0.0;
160  if ( std::fabs( inclination ) < tolerance )
161  {
162  longitudeOfAscendingNode = std::numeric_limits< Real >::quiet_NaN( );
163  }
164  else
165  {
166  const Real cosLongitudeOfAscendingNode = ascendingNodeUnitVector[ 0 ];
167  const Real sinLongitudeOfAscendingNode = ascendingNodeUnitVector[ 1 ];
168  const Real longitudeOfAscendingNodeShifted = std::atan2( sinLongitudeOfAscendingNode,
169  cosLongitudeOfAscendingNode );
170  longitudeOfAscendingNode = longitudeOfAscendingNodeShifted > 0.0
171  ? longitudeOfAscendingNodeShifted
172  : longitudeOfAscendingNodeShifted + 2.0 * pi;
173  }
174 
175  Real argumentOfPeriapsis = 0.0;
176  if ( std::fabs( eccentricity ) < tolerance )
177  {
178  argumentOfPeriapsis = std::numeric_limits< Real >::quiet_NaN( );
179  }
180  else
181  {
182  const Real cosArgumentOfPeriapsis
183  = ascendingNodeUnitVector[ 0 ] * eccentricityVector[ 0 ] / eccentricity
184  + ascendingNodeUnitVector[ 1 ] * eccentricityVector[ 1 ] / eccentricity
185  + ascendingNodeUnitVector[ 2 ] * eccentricityVector[ 2 ] / eccentricity;
186  const Real sinArgumentOfPeriapsis
187  = ( ( ascendingNodeUnitVector[ 1 ] * eccentricityVector[ 2 ]
188  - ascendingNodeUnitVector[ 2 ] * eccentricityVector[ 1 ] ) / eccentricity )
189  * angularMomentumUnitVector[ 0 ]
190  + ( ( ascendingNodeUnitVector[ 2 ] * eccentricityVector[ 0 ]
191  - ascendingNodeUnitVector[ 0 ] * eccentricityVector[ 2 ] ) / eccentricity )
192  * angularMomentumUnitVector[ 1 ]
193  + ( ( ascendingNodeUnitVector[ 0 ] * eccentricityVector[ 1 ]
194  - ascendingNodeUnitVector[ 1 ] * eccentricityVector[ 0 ] ) / eccentricity )
195  * angularMomentumUnitVector[ 2 ];
196  const Real argumentOfPeriapsisShifted = std::atan2( sinArgumentOfPeriapsis,
197  cosArgumentOfPeriapsis );
198  argumentOfPeriapsis = argumentOfPeriapsisShifted > 0.0
199  ? argumentOfPeriapsisShifted
200  : argumentOfPeriapsisShifted + 2.0 * pi;
201  }
202 
203  Real trueAnomaly = 0.0;
204  if ( std::fabs( eccentricity ) < tolerance )
205  {
206  if ( std::fabs( inclination ) < tolerance )
207  {
208  trueAnomaly = ( velocity[ 0 ] < 0.0 )
209  ? std::acos( position[ 0 ] / positionNorm )
210  : 2.0 * pi - std::acos( position[ 0 ] / positionNorm );
211  }
212  else
213  {
214  trueAnomaly = std::acos( position[ 0 ] / positionNorm * ascendingNodeUnitVector[ 0 ]
215  + position[ 1 ] / positionNorm * ascendingNodeUnitVector[ 1 ]
216  + position[ 2 ] / positionNorm * ascendingNodeUnitVector[ 2 ] );
217  }
218  }
219  else
220  {
221  const Real cosTrueAnomaly
222  = eccentricityVector[ 0 ] * position[ 0 ] / ( eccentricity * positionNorm )
223  + eccentricityVector[ 1 ] * position[ 1 ] / ( eccentricity * positionNorm )
224  + eccentricityVector[ 2 ] * position[ 2 ] / ( eccentricity * positionNorm );
225  const Real sinTrueAnomaly
226  = ( ( eccentricityVector[ 1 ] * position[ 2 ]
227  - eccentricityVector[ 2 ] * position[ 1 ] ) / ( eccentricity * positionNorm ) )
228  * angularMomentumUnitVector[ 0 ]
229  + ( ( eccentricityVector[ 2 ] * position[ 0 ]
230  - eccentricityVector[ 0 ] * position[ 2 ] ) / ( eccentricity * positionNorm ) )
231  * angularMomentumUnitVector[ 1 ]
232  + ( ( eccentricityVector[ 0 ] * position[ 1 ]
233  - eccentricityVector[ 1 ] * position[ 0 ] ) / ( eccentricity * positionNorm ) )
234  * angularMomentumUnitVector[ 2 ];
235  const Real trueAnomalyShifted = std::atan2( sinTrueAnomaly, cosTrueAnomaly );
236  trueAnomaly = trueAnomalyShifted > 0.0 ? trueAnomalyShifted : trueAnomalyShifted + 2.0 * pi;
237  }
238 
239  Vector6 keplerianElements = cartesianElements;
240  keplerianElements[ 0 ] = semiMajorAxis;
241  keplerianElements[ 1 ] = eccentricity;
242  keplerianElements[ 2 ] = inclination;
243  keplerianElements[ 3 ] = argumentOfPeriapsis;
244  keplerianElements[ 4 ] = longitudeOfAscendingNode;
245  keplerianElements[ 5 ] = trueAnomaly;
246  return keplerianElements;
247 }

279 template< typename Real, typename Vector6 >
280 Vector6 convertKeplerianToCartesianElements(
281  const Vector6& keplerianElements, const Real gravitationalParameter,
282  const Real tolerance = 10.0 * std::numeric_limits< Real >::epsilon( ) )
283 {
284  Vector6 cartesianElements = keplerianElements;
285 
286  const Real semiMajorAxis = keplerianElements[ semiMajorAxisIndex ];
287  const Real eccentricity = keplerianElements[ eccentricityIndex ];
288  const Real inclination = keplerianElements[ inclinationIndex ];
289  const Real argumentOfPeriapsis = keplerianElements[ argumentOfPeriapsisIndex ];
290  const Real longitudeOfAscendingNode = keplerianElements[ longitudeOfAscendingNodeIndex ];
291  const Real trueAnomaly = keplerianElements[ trueAnomalyIndex ];
292 
293  // Pre-compute sines and cosines of angles for efficient computation.
294  const Real cosineOfInclination = std::cos( inclination );
295  const Real sineOfInclination = std::sin( inclination );
296  const Real cosineOfArgumentOfPeriapsis = std::cos( argumentOfPeriapsis );
297  const Real sineOfArgumentOfPeriapsis = std::sin( argumentOfPeriapsis );
298  const Real cosineOfLongitudeOfAscendingNode = std::cos( longitudeOfAscendingNode );
299  const Real sineOfLongitudeOfAscendingNode = std::sin( longitudeOfAscendingNode );
300  const Real cosineOfTrueAnomaly = std::cos( trueAnomaly );
301  const Real sineOfTrueAnomaly = std::sin( trueAnomaly );
302 
303  // Compute semi-latus rectum in the case the orbit is not a parabola.
304  Real semiLatusRectum = 0.0;
305  if ( std::fabs( eccentricity - 1.0 ) > tolerance )
306  {
307  semiLatusRectum = semiMajorAxis * ( 1.0 - eccentricity * eccentricity );
308  }
309 
310  // Else set the semi-latus rectum as the first element in the vector of Keplerian elements.
311  else
312  {
313  semiLatusRectum = keplerianElements[ 0 ];
314  }
315 
316  // Compute the magnitude of the orbital radius, measured from the focal point.
317  const Real radiusMagnitude = semiLatusRectum / ( 1.0 + eccentricity * cosineOfTrueAnomaly );
318 
319  // Define position and velocity in the perifocal coordinate system.
320  const Real xPositionPerifocal = radiusMagnitude * cosineOfTrueAnomaly;
321  const Real yPositionPerifocal = radiusMagnitude * sineOfTrueAnomaly;
322  const Real xVelocityPerifocal
323  = -std::sqrt( gravitationalParameter / semiLatusRectum ) * sineOfTrueAnomaly;
324  const Real yVelocityPerifocal
325  = std::sqrt( gravitationalParameter / semiLatusRectum )
326  * ( eccentricity + cosineOfTrueAnomaly );
327 
328  // Compute scalar components of rotation matrix to rotate from periforcal to Earth-Centered
329  // Inertial (ECI) frame.
330  const Real rotationMatrixComponent11
331  = ( cosineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis
332  - sineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis * cosineOfInclination );
333  const Real rotationMatrixComponent12
334  = ( -cosineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis
335  -sineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis * cosineOfInclination );
336 
337  const Real rotationMatrixComponent21
338  = ( sineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis
339  + cosineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis * cosineOfInclination );
340  const Real rotationMatrixComponent22
341  = ( -sineOfLongitudeOfAscendingNode * sineOfArgumentOfPeriapsis
342  + cosineOfLongitudeOfAscendingNode * cosineOfArgumentOfPeriapsis * cosineOfInclination );
343 
344  const Real rotationMatrixComponent31 = ( sineOfArgumentOfPeriapsis * sineOfInclination );
345  const Real rotationMatrixComponent32 = ( cosineOfArgumentOfPeriapsis * sineOfInclination );
346 
347  // Compute Cartesian position and velocities.
348  cartesianElements[ xPositionIndex ] = rotationMatrixComponent11 * xPositionPerifocal
349  + rotationMatrixComponent12 * yPositionPerifocal;
350 
351  cartesianElements[ yPositionIndex ] = rotationMatrixComponent21 * xPositionPerifocal
352  + rotationMatrixComponent22 * yPositionPerifocal;
353 
354  cartesianElements[ zPositionIndex ] = rotationMatrixComponent31 * xPositionPerifocal
355  + rotationMatrixComponent32 * yPositionPerifocal;
356 
357  cartesianElements[ xVelocityIndex ] = rotationMatrixComponent11 * xVelocityPerifocal
358  + rotationMatrixComponent12 * yVelocityPerifocal;
359 
360  cartesianElements[ yVelocityIndex ] = rotationMatrixComponent21 * xVelocityPerifocal
361  + rotationMatrixComponent22 * yVelocityPerifocal;
362 
363  cartesianElements[ zVelocityIndex ] = rotationMatrixComponent31 * xVelocityPerifocal
364  + rotationMatrixComponent32 * yVelocityPerifocal;
365 
366  return cartesianElements;
367 }

383 template< typename Real >
384 Real convertTrueAnomalyToEllipticalEccentricAnomaly( const Real trueAnomaly,
385  const Real eccentricity )
386 {
387  if ( eccentricity >= 1.0 || eccentricity < 0.0 )
388  {
389  throw std::runtime_error( "ERROR: Eccentricity is non-elliptical!" );
390  }
391 
392  // Compute sine and cosine of eccentric anomaly.
393  const Real sineOfEccentricAnomaly
394  = std::sqrt( 1.0 - std::pow( eccentricity, 2.0 ) )
395  * std::sin( trueAnomaly ) / ( 1.0 + eccentricity * std::cos( trueAnomaly ) );
396  const Real cosineOfEccentricAnomaly
397  = ( eccentricity + std::cos( trueAnomaly ) )
398  / ( 1.0 + eccentricity * std::cos( trueAnomaly ) );
399 
400  // Return elliptical eccentric anomaly.
401  return std::atan2( sineOfEccentricAnomaly, cosineOfEccentricAnomaly );
402 }

419 template< typename Real >
420 Real convertTrueAnomalyToHyperbolicEccentricAnomaly( const Real trueAnomaly,
421  const Real eccentricity )
422 {
423  if ( eccentricity <= 1.0 )
424  {
425  throw std::runtime_error( "ERROR: Eccentricity is non-hyperbolic!" );
426  }
427 
428  // Compute hyperbolic sine and hyperbolic cosine of hyperbolic eccentric anomaly.
429  const Real hyperbolicSineOfHyperbolicEccentricAnomaly
430  = std::sqrt( std::pow( eccentricity, 2.0 ) - 1.0 )
431  * std::sin( trueAnomaly ) / ( 1.0 + std::cos( trueAnomaly ) );
432 
433  const Real hyperbolicCosineOfHyperbolicEccentricAnomaly
434  = ( std::cos( trueAnomaly ) + eccentricity ) / ( 1.0 + std::cos( trueAnomaly ) );
435 
436  // Return hyperbolic eccentric anomaly.
437  // The inverse hyperbolic tangent is computed here manually, since the atanh() function is not
438  // available in older C++ compilers.
439  const Real angle
440  = hyperbolicSineOfHyperbolicEccentricAnomaly
441  / hyperbolicCosineOfHyperbolicEccentricAnomaly;
442  return 0.5 * ( std::log( 1.0 + angle ) - std::log( 1.0 - angle ) );
443 }

465 Real convertTrueAnomalyToEccentricAnomaly( const Real trueAnomaly, const Real eccentricity )

785 template< typename Real, typename Integer >
786 Real convertEllipticalMeanAnomalyToEccentricAnomaly(
787  const Real eccentricity,
788  const Real meanAnomaly,
789  const Real rootFindingTolerance = 1.0e-3 * std::numeric_limits< Real >::epsilon( ),
790  const Integer maximumIterations = 100 )
791 {
792  const Real pi = 3.14159265358979323846;
793 
794  // Check if eccentricity is invalid and throw an error if true.
795  if ( eccentricity < 0.0 )
796  {
797  throw std::runtime_error( "ERROR: Eccentricity is negative!" );
798  }
799  else if ( eccentricity > ( 1.0 - 1.0e-11 ) )
800  {
801  throw std::runtime_error( "ERROR: Eccentricity is non-elliptical!" );
802  }
803 
804  // Set mean anomaly to domain between 0 and 2pi.
805  Real meanAnomalyShifted = std::fmod( meanAnomaly, 2.0 * pi );
806  if ( meanAnomalyShifted < 0.0 )
807  {
808  meanAnomalyShifted += 2.0 * pi;
809  }
810 
811  Real eccentricAnomaly = std::numeric_limits< Real >::quiet_NaN( );
812 
813  // Check if orbit is elliptical, and not near-parabolic.
814  if ( eccentricity < ( 1.0 - 1.0e-11 ) )
815  {
816  // Set the initial guess for the eccentric anomaly.
817  // !!!!!!!!!!!!! IMPORTANT !!!!!!!!!!!!!
818  // If this scheme is changed, please run a very extensive test suite. The Newton-Raphson
819  // root-finding scheme tends to be chaotic for very specific combinations of mean anomaly
820  // and eccentricity. Various random tests of 100,000,000 samples were done to verify the
821  // default scheme for the initial guess. (Musegaas, 2012).
822  Real initialGuess = 0.0;
823  if ( meanAnomalyShifted > pi )
824  {
825  initialGuess = meanAnomalyShifted - eccentricity;
826  }
827  else
828  {
829  initialGuess = meanAnomalyShifted + eccentricity;
830  }
831 
832  // Execute Newton-Raphson root-finding algorithm.
833  eccentricAnomaly = initialGuess;
834  for ( int i = 0; i < maximumIterations + 1; ++i )
835  {
836  if ( i == maximumIterations )
837  {
838  throw std::runtime_error(
839  "ERROR: Maximum iterations for Newton-Raphson root-finding exceeded!" );
840  }
841 
842  const Real nextEccentricAnomaly
843  = eccentricAnomaly
844  - computeEllipticalKeplerFunction( eccentricAnomaly,
845  eccentricity,
846  meanAnomalyShifted )
847  / computeFirstDerivativeEllipticalKeplerFunction( eccentricAnomaly,
848  eccentricity );
849 
850  const Real eccentricAnomalyDifference = eccentricAnomaly - nextEccentricAnomaly;
851 
852  eccentricAnomaly = nextEccentricAnomaly;
853 
854  if ( eccentricAnomalyDifference < rootFindingTolerance )
855  {
856  break;
857  }
858  }
859  }
860  else
861  {
862  throw std::runtime_error( "ERROR: Eccentricity is non-elliptical!" );
863  }
864 
865  // Return eccentric anomaly.
866  return eccentricAnomaly;
867 }
